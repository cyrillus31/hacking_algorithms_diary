func findMaxConsecutiveOnes(nums []int) int {
    count := 0
    maxCount := 0
    /*
    1 1 0 1 1
    1 2 0 1 2
    */ 
    for _, num := range nums {
        count = count*num + num
        if count > maxCount {
            maxCount = count
        }
    }
    return maxCount
}
